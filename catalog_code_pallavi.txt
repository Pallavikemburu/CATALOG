function decodeValue(value, base) {    return parseInt(value, base);}function lagrangeInterpolation(points, x) {    let total = 0;    const n = points.length;    for (let i = 0; i < n; i++) {        let xi = points[i][0];        let yi = points[i][1];        let li = 1;        for (let j = 0; j < n; j++) {            if (j !== i) {                li *= (x - points[j][0]) / (xi - points[j][0]);            }        }        total += yi * li;    }    return total;}function findConstantTerm(data) {    const keys = Object.keys(data).filter(key => key !== 'keys');    const k = data.keys.k;    const points = keys.map(key => {        const x = parseInt(key, 10);        const yEncoded = data[key].value;        const base = parseInt(data[key].base, 10);        const y = decodeValue(yEncoded, base);        return [x, y];    });    // Sort points by x to ensure correct processing order    points.sort((a, b) => a[0] - b[0]);    // Use the first k points to find the constant term using Lagrange interpolation    const selectedPoints = points.slice(0, k);    return lagrangeInterpolation(selectedPoints, 0);}function main() {    const testData1 = {        "keys": {            "n": 4,            "k": 3        },        "1": {            "base": "10",            "value": "4"        },        "2": {            "base": "2",            "value": "111"        },        "3": {            "base": "10",            "value": "12"        },        "6": {            "base": "4",            "value": "213"        }    };    const testData2 = {        "keys": {            "n": 10,            "k": 7        },        "1": {            "base": "6",            "value": "13444211440455345511"        },        "2": {            "base": "15",            "value": "aed7015a346d63"        },        "3": {            "base": "15",            "value": "6aeeb69631c227c"        },        "4": {            "base": "16",            "value": "e1b5e05623d881f"        },        "5": {            "base": "8",            "value": "316034514573652620673"        },        "6": {            "base": "3",            "value": "2122212201122002221120200210011020220200"        },        "7": {            "base": "3",            "value": "20120221122211000100210021102001201112121"        },        "8": {            "base": "6",            "value": "20220554335330240002224253"        },        "9": {            "base": "12",            "value": "45153788322a1255483"        },        "10": {            "base": "7",            "value": "1101613130313526312514143"        }    };    const secret1 = findConstantTerm(testData1);    const secret2 = findConstantTerm(testData2);    console.log(`c for testcase 1: ${secret1}`);    console.log(`c for testcase 2: ${secret2}`);}main();
